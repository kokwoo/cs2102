CREATE TABLE users  (
  userId INT PRIMARY KEY,
  name VARCHAR(64),
  password VARCHAR(255),
  address VARCHAR(255),
  role VARCHAR(16),
  lastLogin TIMESTAMP
);

CREATE TABLE item (
  itemId INTEGER PRIMARY KEY,
  type VARCHAR(64),
  name VARCHAR(255),
  price NUMERIC,
  avaliability BOOLEAN,
  date_posted DATE now()
);

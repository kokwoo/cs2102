CREATE TABLE users  (
  userid VARCHAR(64) PRIMARY KEY,
  name VARCHAR(64),
  password CHAR(64),
  address VARCHAR(255),
  role VARCHAR(16),
  lastlogin TIMESTAMP
);

CREATE TABLE item (
  itemId INTEGER PRIMARY KEY,
  type VARCHAR(64),
  name VARCHAR(255),
  price NUMERIC,
  avaliability BOOLEAN,
  date_posted DATE now()
);
